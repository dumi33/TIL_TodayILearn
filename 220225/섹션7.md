```py
while True :
    if L == (n//2) :

    size = len(q)

            y = tmp[1]+ dy[j]
```

### 문제 12 - 단지번호 붙이기 

- bfs 풀이 
              
```py
from collections import deque

dx = [0,-1,0,1]
dy = [1,0,-1,0]

if __name__ == "__main__" :

    n = int(input())
    board = [list(map(int,input())) for _ in range(n)]
    cnt = []# 단지의 개수 
    sum = 0
    dq = deque()
    for i in range(n) :
        for j in range(n) :
            if board[i][j] == 1 :
                board[i][j] = 0
                dq.append((i,j))
                sum = 1 
                while dq :
                    x,y = dq.popleft()
                    for k in range(4):
                        nx = x+dx[k]
                        ny = y+dy[k]
                        if 0<=nx<n and 0<=ny<n and board[nx][ny] == 1 :
                            dq.append((nx,ny))
                            sum+=1 
                            board[nx][ny] =0
                cnt.append(sum)
                 

    print(len(cnt))
    for i in sorted(cnt) :
```
