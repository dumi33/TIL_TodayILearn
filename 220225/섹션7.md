### 문제 2 - 휴가(DFS)


- 내가 만든 코드 
```py
def dfs(L,money) :
    global max_money 
    if L > n : # 날이 n보다 커버리면 
        return 
    if L == n : # 모든 날이 지나갔을 때
        if money > max_money : 
            max_money = money
    else :
        dfs(L+graph[L][0],money+graph[L][1]) # 상담을 했을 때 
        dfs(L+1,money) # 상담을 하지않았을 때 
                

if __name__ == "__main__" :
    n= int(input())
    max_money = 0
    graph = [] 
    for _ in range(n) :
        d,m = map(int,input().split()) #  day : 소요 날짜, money
        graph.append((d,m))  # d만큼 시간이 소요되고, m 만큼 보수를 받는다.
    dfs(0, 0) # 0 카운슬러(level), money : 0점이다. 
    print(max_money)
 ```
